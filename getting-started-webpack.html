Getting Started

With NPM and Webpack

The best way to get started with Webpack and React is with Create React App.


npx create-react-app my-app
cd my-app
npm start

From here, you can modify src/App.js to use ylem and see the live-reload in the browser.


import React, { Component } from 'react';
import logo from './logo.svg';
import './App.css';
import { Component } from 'ylem';

class App extends Component { // ðŸ‘€
  constructor(props) {
    super(props);
    this.state = { count: 0 };
  }

  increment = () => {
    this.state.count++; // ðŸ‘€
  }

  render() {
    return (
      <div className="App">
        <header className="App-header">
          <img src={logo} className="App-logo" alt="logo" />
          <p>
            Count: {this.state.count}
          </p>
          <a className="App-link" onClick={this.increment}>+1</a>
        </header>
      </div>
    );
  }
}

export default App;
